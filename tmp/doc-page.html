      [HTML]
      e.g., "Selector"
        O: elem-root [Default], elem-y [Default], elem-yotta (Warning: elem-yottayotta(X)), elem-yb
        X: elem-yottabyte, elem-yottaByte, elem-yottayotta

      [JS] 
      e.g., "Variable"
        O: elemY [Default] (Warning: Last character only, elemYTitle(X)), elemYottaTitle [Default], yotta [Default] (Warning: Local function only), elemYotta, yElem (Warning: YElem(X))

  [FWS] Facooya Web Standard updated {
    [HTML], [CSS], [JS] Selector {
      id -> class (Feasible)
      [HTML]
      e.g., "Selector"
        O: class [Default] ([JS] Reference (querySelector)), id [Alternate] ([JS] Reference (getElement))

      [CSS]
      e.g., "Style"
        O: .root [Default] (class selector only)
        X: #root (id selector)

      [JS]
      e.g., "Code"
        O: querySelector(".root") [Default], getElementById("root") (root.value)
        X: getElementsByClassName("root") (root[0].value)

      Overview: Avoid the use of the "id" attribute, instead use "class" attribute.
    }
    [HTML], [CSS] PseudoElement {
      [CSS] Pseudo element -> [HTML] Element
        e.g., [CSS] .element::after -> [HTML] class="element-after"

      Overview: Avoid the use of pseudo elements, instead add elements.
    }
    [CSS] Grid {
      flex -> grid (Feasible)
      width -> grid-template-columns, grid-auto-columns (Feasible)
      hieght -> grid-template-rows, grid-auto-rows (Feasible)
      margin, padding, ... -> gap (Feasible)
    }
    [JS] Element {
      querySelector, querySelectorAll -> getElementById, getElementsByClassName (Feasible)
      Expetion: FWS.Selector
      e.g., "Selector"
        O: getElementsByClassName [Default], getElementById, querySelector (FWS.Selector)
        X: querySelectorAll

      Overview: Avoid the use of the "query", instead use "getElement".
    }
    [CSS], [JS] Event {
      [CSS] :hover -> [JS] eventListener("event", handler) (All)

          addEventListener("mouseenter", mouseEnter);

      Reference: Performance updated
